## 7. WebSocket 浜や簰

```javascript
// websocket.js
class AstraionWebSocket {
  constructor() {
    this.ws = null
    this.messageId = 0
  }
  
  connect(token) {
    this.ws = new WebSocket(`ws://${host}/api/v1/ai/chat?token=${token}`)
    
    this.ws.onmessage = (event) => {
      const msg = JSON.parse(event.data)
      
      switch (msg.type) {
        case 'thinking':
          this.showThinking(msg.content)
          break
          
        case 'render':
          this.renderComponent({
            component: msg.component,   // table | form | detail | confirm | approval | init
            props: msg.props,
            id: ++this.messageId
          })
          break
          
        case 'message':
          this.showMessage(msg.content)
          break
      }
    }
  }
  
  sendMessage(content) {
    this.ws.send(JSON.stringify({ type: 'user_message', content }))
  }
}
```
